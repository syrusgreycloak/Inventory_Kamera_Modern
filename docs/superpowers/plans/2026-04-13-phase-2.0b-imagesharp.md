**Goal:** Replace the GDI+ (System.Drawing) image processing backend with SixLabors.ImageSharp. The `IImageProcessor` interface stays System.Drawing at its boundaries (since `IScreenCapture` returns `Bitmap` and the UI requires `Bitmap`), but the internal implementation switches to ImageSharp with conversions at the entry/exit points. Accord.Imaging (abandoned, unused) is removed.

**Architecture:** Create a new `ImageSharpProcessor` class in `InventoryKamera/Infrastructure/` that implements `IImageProcessor`. It converts incoming `System.Drawing.Bitmap` to `SixLabors.ImageSharp.Image<Rgba32>`, processes it using ImageSharp operations, then converts back to `Bitmap` on return. `WinFormsImageProcessor` is kept as a fallback until tests confirm equivalence, then removed. `GenshinProcesor` image methods are eventually removable but are not deleted in this plan.

**Tech Stack:** SixLabors.ImageSharp 3.1.x, System.Drawing.Common 8.0.0 (stays in Core interfaces for now), xUnit 2.9.x

**Prerequisite:** Phase 2.0a must be complete. All scraper bodies must be calling `_imageProcessor.*` rather than `GenshinProcesor.*`.

- [ ] **Step 1: Write the conversion helpers**

Converting between System.Drawing.Bitmap and SixLabors.ImageSharp.Image<Rgba32> requires reading raw pixel bytes. Both types expose unsafe/raw pixel access.

```csharp
using System;
using System.Collections.Generic;
using System.Drawing;
using System.Drawing.Imaging;
using System.Runtime.InteropServices;
using SixLabors.ImageSharp;
using SixLabors.ImageSharp.PixelFormats;
using SixLabors.ImageSharp.Processing;
using SixLabors.ImageSharp.Processing.Processors.Filters;

namespace InventoryKamera.Infrastructure
{
    internal class ImageSharpProcessor : IImageProcessor
    {
        // --- Conversion helpers ---

        private static Image<Rgba32> ToImageSharp(Bitmap bmp)
        {
            BitmapData data = bmp.LockBits(
                new System.Drawing.Rectangle(0, 0, bmp.Width, bmp.Height),
                ImageLockMode.ReadOnly, System.Drawing.Imaging.PixelFormat.Format32bppArgb);
            try
            {
                int stride = data.Stride;
                byte[] pixels = new byte[stride * bmp.Height];
                Marshal.Copy(data.Scan0, pixels, 0, pixels.Length);

                var img = new Image<Rgba32>(bmp.Width, bmp.Height);
                img.ProcessPixelRows(accessor =>
                {
                    for (int y = 0; y < accessor.Height; y++)
                    {
                        var row = accessor.GetRowSpan(y);
                        for (int x = 0; x < accessor.Width; x++)
                        {
                            int pos = y * stride + x * 4;
                            // BGRA in GDI+, RGBA in ImageSharp
                            row[x] = new Rgba32(pixels[pos + 2], pixels[pos + 1], pixels[pos + 0], pixels[pos + 3]);
                        }
                    }
                });
                return img;
            }
            finally
            {
                bmp.UnlockBits(data);
            }
        }

        private static Bitmap ToBitmap(Image<Rgba32> img)
        {
            var bmp = new Bitmap(img.Width, img.Height, System.Drawing.Imaging.PixelFormat.Format32bppArgb);
            BitmapData data = bmp.LockBits(
                new System.Drawing.Rectangle(0, 0, bmp.Width, bmp.Height),
                ImageLockMode.WriteOnly, System.Drawing.Imaging.PixelFormat.Format32bppArgb);
            try
            {
                int stride = data.Stride;
                byte[] pixels = new byte[stride * bmp.Height];
                img.ProcessPixelRows(accessor =>
                {
                    for (int y = 0; y < accessor.Height; y++)
                    {
                        var row = accessor.GetRowSpan(y);
                        for (int x = 0; x < accessor.Width; x++)
                        {
                            int pos = y * stride + x * 4;
                            pixels[pos + 0] = row[x].B;
                            pixels[pos + 1] = row[x].G;
                            pixels[pos + 2] = row[x].R;
                            pixels[pos + 3] = row[x].A;
                        }
                    }
                });
                Marshal.Copy(pixels, 0, data.Scan0, pixels.Length);
                return bmp;
            }
            finally
            {
                bmp.UnlockBits(data);
            }
        }
```

- [ ] **Step 6: Implement `SetContrast`**

ImageSharp's contrast filter uses a -1 to 1 float scale, not the 0–255 GDI+ convention. Map accordingly using the same formula as GenshinProcesor.

```csharp
        public Bitmap SetContrast(Bitmap bitmap, double contrast)
        {
            // Replicate GenshinProcesor formula: factor = (259*(c+255)) / (255*(259-c))
            // Apply as a pixel-by-pixel LUT for exact equivalence
            int c = (int)contrast;
            double factor = (259.0 * (c + 255)) / (255.0 * (259 - c));
            byte[] lut = new byte[256];
            for (int i = 0; i < 256; i++)
                lut[i] = (byte)Math.Max(0, Math.Min(255, (int)(factor * (i - 128) + 128)));

            using (var img = ToImageSharp(bitmap))
            {
                img.ProcessPixelRows(accessor =>
                {
                    for (int y = 0; y < accessor.Height; y++)
                    {
                        var row = accessor.GetRowSpan(y);
                        for (int x = 0; x < accessor.Width; x++)
                        {
                            var px = row[x];
                            row[x] = new Rgba32(lut[px.R], lut[px.G], lut[px.B], px.A);
                        }
                    }
                });
                return ToBitmap(img);
            }
        }
```

```csharp
        private static byte[] BuildGammaLut(double gamma)
        {
            byte[] lut = new byte[256];
            for (int i = 0; i < 256; i++)
                lut[i] = (byte)Math.Min(255, (int)(255.0 * Math.Pow(i / 255.0, 1.0 / gamma) + 0.5));
            return lut;
        }
```

- [ ] **Step 12: Implement `KirschEdgeDetect`**

ImageSharp does not have Kirsch built-in. Port the existing pixel-by-pixel implementation using `ProcessPixelRows`. The kernel arrays are identical to GenshinProcesor.

```csharp
        public Bitmap KirschEdgeDetect(Bitmap source)
        {
            int[][][] kernels = new int[][][]
            {
                new int[][] { new int[] {  5,  5,  5 }, new int[] { -3,  0, -3 }, new int[] { -3, -3, -3 } },
                new int[][] { new int[] {  5,  5, -3 }, new int[] {  5,  0, -3 }, new int[] { -3, -3, -3 } },
                new int[][] { new int[] {  5, -3, -3 }, new int[] {  5,  0, -3 }, new int[] {  5, -3, -3 } },
                new int[][] { new int[] { -3, -3, -3 }, new int[] {  5,  0, -3 }, new int[] {  5,  5, -3 } },
                new int[][] { new int[] { -3, -3, -3 }, new int[] { -3,  0, -3 }, new int[] {  5,  5,  5 } },
                new int[][] { new int[] { -3, -3, -3 }, new int[] { -3,  0,  5 }, new int[] { -3,  5,  5 } },
                new int[][] { new int[] { -3, -3,  5 }, new int[] { -3,  0,  5 }, new int[] { -3, -3,  5 } },
                new int[][] { new int[] { -3,  5,  5 }, new int[] { -3,  0,  5 }, new int[] { -3, -3, -3 } },
            };

            using (var src = ToImageSharp(source))
            {
                var result = new Image<Rgba32>(src.Width, src.Height);

                // Read all source pixels into a flat array for random access
                var srcPixels = new Rgba32[src.Width * src.Height];
                src.ProcessPixelRows(accessor =>
                {
                    for (int y = 0; y < accessor.Height; y++)
                    {
                        var row = accessor.GetRowSpan(y);
                        for (int x = 0; x < accessor.Width; x++)
                            srcPixels[y * src.Width + x] = row[x];
                    }
                });

                result.ProcessPixelRows(accessor =>
                {
                    for (int y = 1; y < src.Height - 1; y++)
                    {
                        var row = accessor.GetRowSpan(y);
                        for (int x = 1; x < src.Width - 1; x++)
                        {
                            int maxResponse = 0;
                            foreach (var kernel in kernels)
                            {
                                int response = 0;
                                for (int ky = 0; ky < 3; ky++)
                                    for (int kx = 0; kx < 3; kx++)
                                    {
                                        var px = srcPixels[(y + ky - 1) * src.Width + (x + kx - 1)];
                                        int gray = (px.R + px.G + px.B) / 3;
                                        response += kernel[ky][kx] * gray;
                                    }
                                if (response > maxResponse) maxResponse = response;
                            }
                            byte edge = (byte)Math.Min(255, Math.Max(0, maxResponse));
                            row[x] = new Rgba32(edge, edge, edge, 255);
                        }
                    }
                });

                return ToBitmap(result);
            }
        }
```

```csharp
        public System.Drawing.Color ClosestColor(IList<System.Drawing.Color> colors, System.Drawing.Color target)
        {
            foreach (var c in colors)
                if (CompareColors(c, target)) return c;
            System.Drawing.Color best = colors[0];
            int bestDiff = int.MaxValue;
            foreach (var c in colors)
            {
                int dr = c.R - target.R, dg = c.G - target.G, db = c.B - target.B;
                int diff = dr * dr + dg * dg + db * db;
                if (diff < bestDiff) { bestDiff = diff; best = c; }
            }
            return best;
        }

        public bool CompareColors(System.Drawing.Color a, System.Drawing.Color b)
        {
            return Math.Abs(a.R - b.R) < 10 && Math.Abs(a.G - b.G) < 10 && Math.Abs(a.B - b.B) < 10;
        }

        public bool CompareBitmapsFast(Bitmap bmp1, Bitmap bmp2)
        {
            if (bmp1 == null || bmp2 == null) return false;
            if (object.ReferenceEquals(bmp1, bmp2)) return true;
            if (bmp1.Width != bmp2.Width || bmp1.Height != bmp2.Height) return false;

            using (var img1 = ToImageSharp(bmp1))
            using (var img2 = ToImageSharp(bmp2))
            {
                bool equal = true;
                img1.ProcessPixelRows(img2, (acc1, acc2) =>
                {
                    for (int y = 0; y < acc1.Height && equal; y++)
                    {
                        var row1 = acc1.GetRowSpan(y);
                        var row2 = acc2.GetRowSpan(y);
                        for (int x = 0; x < row1.Length; x++)
                        {
                            if (row1[x].PackedValue != row2[x].PackedValue) { equal = false; break; }
                        }
                    }
                });
                return equal;
            }
        }
```
